1. Even-Odd Check

object EvenOddCheck {
  def main(args: Array[String]): Unit = {
    val number = 15
    if (number % 2 == 0) {
      println(s"$number is even")
    } else {
      println(s"$number is odd")
    }
  }
}

6. Factorial

object Factorial {
  def main(args: Array[String]): Unit = {
    val num = 5
    var factorial = 1
    for (i <- 1 to num) {
      factorial *= i
    }
    println(s"The factorial of $num is $factorial")
  }
}
